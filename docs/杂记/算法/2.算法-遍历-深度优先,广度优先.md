# 算法-DFS-深度优先遍历
- Depth-First Search
```javascript
 const root = document.getElementsByClassName('parent')[0];
    const res = [];
    /**
     * 深度优先遍历
     */
    function deepSearch(node) {
        const className = node.className;
        res.push(className);
        if(node && node.children && node.children.length > 0){
            for(let item of node.children){
                deepSearch(item);
            }
        }
    }
    deepSearch(root);
    console.log(res);
```
# 算法-BFS-广度优先遍历
- Breadth-First Search
```javascript
const root = document.getElementsByClassName('parent')[0];
    const res = [root.className];
    /**
     * 广度优先遍历
     * 每一层的children 依次入栈
     */
    function deepSearch(children = []) {
        let nextChildren = [];
       for(let item of children){
           res.push(item.className);
           if(item.children.length){
               // 将下一层的children 依次入栈
               nextChildren.push(...item.children);
           }
       }
       if(nextChildren.length){
           deepSearch(nextChildren);
       }
    }
    deepSearch(root.children);
    console.log(res);
```
