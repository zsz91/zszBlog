# 逻辑赋值运算符 ES2020
```javascript
 a ??= 2
 a ||= 3 
 a &&= 4
```
2. ||=
```
 a ||= b  等价于 a || (a = b);
```
2. &&=
```
   a &&= b  等价于 a && (a = b);
```

101_475_938.38; // 亿万

const amount = 12345_00; // 1234500
const amount = 123_4500; // 1234500
const amount = 1_234_500; // 1,234,500

0.000_001; // 百万分之一
1e10_000; // 10^10000

//
const binary_literals = 0b1010_0001_1000_0101;

const bigInt_literals = 1_000_000_000_000n;
