# 浏览器网络状态监控
```javascript
navigator.connection = {
    downlink: 10, // 下载速度 Mb/s
    rtt: 150, // 网络延迟 单位 ms
    saveData: false, // 是否开启省流量模式
    effectiveType: "4g" // 连接类型 2g/3g/4g/wifi
};
navigator.onLine = true; // 表示用户在线 false表示离线
window.addEventListener("online", () => {
    console.log("用户在线");
});
window.addEventListener("offline", () => {
    console.log("用户离线");
});
```
